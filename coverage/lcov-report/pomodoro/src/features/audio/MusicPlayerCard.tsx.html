import { useEffect, useMemo, useState } from "react"
import { useTranslation } from "react-i18next"
import { useAudioController } from "@/features/audio/useAudioController"
import { formatDurationLabel } from "@/features/timer/timerEngine"
 
export function MusicPlayerCard() {
  const { t } = useTranslation('audio')
  const {
    playlist,
    currentTrack,
    isPlaying,
    muted,
    volume,
    toggle,
    next,
    previous,
    toggleMute,
    setVolume,
    selectTrack,
    seekTo,
    getCurrentTime,
    getDuration,
  } = useAudioController()
 
  const [position, setPosition] = useState(0)
  const [duration, setDuration] = useState(0)
 
  const coverUrl = currentTrack?.coverUrl
 
  const otherTracks = useMemo(
    () => playlist.filter((track) => track.id !== currentTrack?.id),
    [playlist, currentTrack],
  )
 
  useEffect(() => {
    let frameId = 0
    const update = () => {
      setPosition(getCurrentTime())
      setDuration(getDuration())
      frameId = window.requestAnimationFrame(update)
    }
    frameId = window.requestAnimationFrame(update)
    return () => window.cancelAnimationFrame(frameId)
  }, [getCurrentTime, getDuration, currentTrack?.id, isPlaying])
 
  const durationSeconds = duration || currentTrack?.durationSec || 0
  const progressValue = durationSeconds > 0 ? Math.min(position, durationSeconds) : 0
 
  return (
    <div className="rounded-3xl border border-white/30 bg-[var(--bg-secondary)]/80 p-6 shadow-card backdrop-blur">
      <div className="flex items-center gap-4">
        <div className="h-20 w-20 overflow-hidden rounded-2xl bg-white/20 shadow-inner">
          {coverUrl ? (
            <img src={coverUrl} alt="" className="h-full w-full object-cover" loading="lazy" />
          ) : (
            <div className="flex h-full w-full items-center justify-center text-3xl">🎧</div>
          )}
        </div>
        <div className="flex-1">
          <p className="text-xs uppercase tracking-wide text-[var(--text-secondary)]">{t('music')}</p>
          <h3 className="text-lg font-semibold text-[var(--text-primary)]">{currentTrack?.title || 'Lofi Loop'}</h3>
          <p className="text-sm text-[var(--text-secondary)]">{currentTrack?.artist || 'Various artists'}</p>
        </div>
      </div>
      <div className="mt-6 flex items-center justify-center gap-3">
        <button
          type="button"
          onClick={() => previous()}
          className="rounded-full bg-white/30 p-3 text-sm text-[var(--text-primary)] transition hover:bg-white/50 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-[var(--accent)]"
          aria-label={t('previous')}
        >
          {'<'}
        </button>
        <button
          type="button"
          onClick={() => toggle()}
          className="rounded-full bg-[var(--accent)] px-6 py-2 text-sm font-semibold text-[var(--text-primary)] shadow-card transition hover:bg-[var(--accent)]/80 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-[var(--accent)]"
        >
          {isPlaying ? t('pause') : t('play')}
        </button>
        <button
          type="button"
          onClick={() => next()}
          className="rounded-full bg-white/30 p-3 text-sm text-[var(--text-primary)] transition hover:bg-white/50 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-[var(--accent)]"
          aria-label={t('next')}
        >
          {'>'}
        </button>
      </div>
      <div className="mt-4 flex items-center gap-2 text-xs text-[var(--text-secondary)]">
        <span className="w-10 text-right">
          {formatDurationLabel(Math.floor(progressValue))}
        </span>
        <input
          type="range"
          min={0}
          max={durationSeconds > 0 ? Math.floor(durationSeconds) : 100}
          step={1}
          value={durationSeconds > 0 ? Math.floor(progressValue) : 0}
          onChange={(event) => seekTo(Number(event.target.value))}
          className="flex-1 accent-[var(--accent)]"
          aria-label={t('music')}
        />
        <span className="w-10 text-left">
          {formatDurationLabel(Math.floor(durationSeconds))}
        </span>
      </div>
      <div className="mt-6 flex items-center gap-3">
        <button
          type="button"
          onClick={() => toggleMute()}
          className="rounded-full bg-white/30 px-3 py-2 text-xs font-medium text-[var(--text-primary)] transition hover:bg-white/50 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-[var(--accent)]"
        >
          {muted ? t('unmute') : t('mute')}
        </button>
        <input
          type="range"
          min={0}
          max={100}
          step={1}
          value={Math.round(volume * 100)}
          onChange={(event) => setVolume(Number(event.target.value) / 100)}
          className="flex-1 accent-[var(--accent)]"
          aria-label={t('volume')}
        />
      </div>
      <div className="mt-6 flex flex-wrap gap-2">
        {otherTracks.map((track) => (
          <button
            key={track.id}
            type="button"
            onClick={() => selectTrack(track.id)}
            className={
              'rounded-full border px-3 py-1.5 text-xs transition focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-[var(--accent)] ' +
              (track.id === currentTrack?.id
                ? 'border-transparent bg-[var(--accent)] text-[var(--text-primary)]'
                : 'border-white/40 bg-white/20 text-[var(--text-secondary)] hover:bg-white/40')
            }
          >
            {track.title}
          </button>
        ))}
      </div>
    </div>
  )
}
